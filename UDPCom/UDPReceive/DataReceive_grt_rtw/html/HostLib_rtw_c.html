   22   /* Open the library and initialize the function pointers */
   23   void CreateHostLibrary(const char *libName, void *hl) 
   24   {
   25       HostLibrary *hostLib = (HostLibrary*)hl;
   26       *hostLib->errorMessage = '\0';
   27       memset(hostLib, 0, sizeof(HostLibrary));
   28   #if defined(_WIN32)
   29       hostLib->library = LoadLibrary(libName);
   30       if(hostLib->library == NULL)
   31           sprintf(hostLib->errorMessage, "Could not open library: %s.\nTo run the generated code outside the MATLAB environment, use the packNGo function.", libName);
   32       else {
   33           hostLib->libCreate    = GetProcAddress(hostLib->library, "libCreate");
   34           hostLib->libStart     = GetProcAddress(hostLib->library, "libStart"); 
   35           hostLib->libReset     = GetProcAddress(hostLib->library, "libReset"); 
   36           hostLib->libUpdate    = GetProcAddress(hostLib->library, "libUpdate");
   37           hostLib->libOutputs   = GetProcAddress(hostLib->library, "libOutputs");
   38           hostLib->libTerminate = GetProcAddress(hostLib->library, "libTerminate");
   39           hostLib->libDestroy   = GetProcAddress(hostLib->library, "libDestroy");
   40       }
   41   #elif defined(_VXWORKS_)
   42       hostLib->library = dlopen(libName, RTLD_NOW);
   43       if(hostLib->library == NULL) {
   44           sprintf(hostLib->errorMessage, "Could not open library: %s", libName);
   45       }
   46       else {
   47           hostLib->libCreate    = dlsym(hostLib->library, "libCreate");
   48           hostLib->libStart     = dlsym(hostLib->library, "libStart"); 
   49           hostLib->libReset     = dlsym(hostLib->library, "libReset"); 
   50           hostLib->libUpdate    = dlsym(hostLib->library, "libUpdate");
   51           hostLib->libOutputs   = dlsym(hostLib->library, "libOutputs");
   52           hostLib->libTerminate = dlsym(hostLib->library, "libTerminate");
   53           hostLib->libDestroy   = dlsym(hostLib->library, "libDestroy");
   54       }
   55   #else
   56       hostLib->library = dlopen(libName, RTLD_NOW | RTLD_LOCAL);
   57       if(hostLib->library == NULL) {
   58           sprintf(hostLib->errorMessage, "Could not open library: %s", libName);
   59       }
   60       else {
   61           hostLib->libCreate    = dlsym(hostLib->library, "libCreate");
   62           hostLib->libStart     = dlsym(hostLib->library, "libStart"); 
   63           hostLib->libReset     = dlsym(hostLib->library, "libReset"); 
   64           hostLib->libUpdate    = dlsym(hostLib->library, "libUpdate");
   65           hostLib->libOutputs   = dlsym(hostLib->library, "libOutputs");
   66           hostLib->libTerminate = dlsym(hostLib->library, "libTerminate");
   67           hostLib->libDestroy   = dlsym(hostLib->library, "libDestroy");
   68       }
   69   #endif
   70       if(!*hostLib->errorMessage && 
   71          (!hostLib->libCreate ||
   72           !hostLib->libStart ||
   73           !hostLib->libUpdate ||
   74           !hostLib->libOutputs ||
   75           !hostLib->libTerminate ||
   76           !hostLib->libDestroy)
   77           ) 
   78       {
   79           sprintf(hostLib->errorMessage, "Could not determine function entry points in %s", libName);
   80       }
   81       if(*hostLib->errorMessage) {
   82           if (hostLib->library) DestroyHostLibrary(hostLib);
   83           fprintf(stderr,"%s",hostLib->errorMessage);
   84       }
   85   }

   87   /* Close the library */
   88   void DestroyHostLibrary(void *hl) 
   89   {
   90       HostLibrary *hostLib = (HostLibrary*)hl;
   91       if(hostLib->library) {
   92   #if defined(_WIN32)
   93           FreeLibrary(hostLib->library);
   94   #else
   95           dlclose(hostLib->library);
   96   #endif
   97           /* Clear the structure (except for any possible error message). */
   98   		memset(hostLib, 0, sizeof(HostLibrary) - MAX_ERR_MSG_LEN);
   99       }
  100   }

  178   void CreateTargetUDPInterface(void *hl) 
  179   {
  180       HostLibrary *hostLib = (HostLibrary*)hl;
  181       *hostLib->errorMessage = '\0';
  182       memset(hostLib, 0, sizeof(HostLibrary));
  183   
  184       /* Indicate that the structure is valid for future function calls. */    
  185       hostLib->library      = (void*) 1;
  186       
  187       /* Set the function pointers to point to target UDP implementation. */    
  188       hostLib->libCreate    = (void*) TGT_libCreate;
  189       hostLib->libStart     = (void*) TGT_libStart; 
  190       hostLib->libReset     = (void*) NULL; 
  191       hostLib->libUpdate    = (void*) TGT_libUpdate;
  192       hostLib->libOutputs   = (void*) TGT_libOutputs;
  193       hostLib->libTerminate = (void*) TGT_libTerminate;
  194       hostLib->libDestroy   = (void*) TGT_libDestroy;
  195   }

  199   {
  200       HostLibrary *hostLib = (HostLibrary*)hl;
  201       if(hostLib->library) {
  202           /* Clear the structure (except for any possible error message). */
  203   		memset(hostLib, 0, sizeof(HostLibrary) - MAX_ERR_MSG_LEN);
  204       }
  205   }

  208   void TGT_libCreate(void **device_, char *err, NetworkInputOrOutput io, 
  209          const char *localURL, int localPort, const char *remoteURL, 
  210          int remotePort, int bufferSize, int dataTypeSize, int blockingTime)
  211   {
  212       int flags;
  213       struct timeval tv;
  214       TgtSockStructT *device;
  215       device = (TgtSockStructT *) malloc (sizeof(TgtSockStructT)); 
  216       
  217       bzero(&device->remotesockaddr, sizeof(device->remotesockaddr));
  218       device->remotesockaddr.sin_family = AF_INET;
  219       device->remotesockaddr.sin_port = htons(remotePort);
  220       inet_aton(remoteURL, &device->remotesockaddr.sin_addr);
  221       
  222       bzero(&device->localsockaddr, sizeof(device->localsockaddr));
  223       device->localsockaddr.sin_family = AF_INET;
  224       device->localsockaddr.sin_port = htons(localPort);
  225       inet_aton(localURL, &device->localsockaddr.sin_addr); 
  226       
  227       device->sockfd = socket(AF_INET, SOCK_DGRAM, 0);    
  228       if (device->sockfd<0)
  229           printf ("error creating socket\n");
  230   
  231       setsockopt(((TgtSockStructT *) device)->sockfd, SOL_SOCKET, 
  232                   SO_RCVBUF, &bufferSize, dataTypeSize);
  233   
  234       if(blockingTime == 0) { 
  235           /* set non-blocking socket option */
  236           flags = fcntl(((TgtSockStructT *) device)->sockfd, F_GETFL, 0);
  237           fcntl(((TgtSockStructT *) device)->sockfd, F_SETFL, flags | O_NONBLOCK);      
  238       }
  239       else if(blockingTime != MIN_int32_T) 
  240       { 
  241           /* if not set to inf in the block mask */
  242           tv.tv_sec  = (long) (blockingTime/1000);
  243           tv.tv_usec = (long) (blockingTime % 1000);    
  244           setsockopt(((TgtSockStructT *) device)->sockfd, SOL_SOCKET, 
  245                   SO_RCVTIMEO, &tv, sizeof(tv));
  246       }       
  247   
  248       device->dataTypeSize = dataTypeSize;
  249   
  250       *device_ = device;
  251   
  252       err[0] = '\0';
  253   }

  256   void TGT_libStart(void *device, char *err) 
  257   {
  258       if(device) {
  259           bind(((TgtSockStructT *) device)->sockfd, 
  260                (struct sockaddr *) &((TgtSockStructT *) device)->localsockaddr, 
  261                sizeof(((TgtSockStructT *) device)->localsockaddr));
  262       }
  263   }

  266   void TGT_libUpdate(void *device, char *err, const void *src, int nSamples)
  267   {
  268       int n;  
  269       if(device) {   
  270           n = sendto(((TgtSockStructT *) device)->sockfd, src, 
  271                      ((TgtSockStructT *) device)->dataTypeSize * nSamples, 0, 
  272                      (struct sockaddr *) &((TgtSockStructT *) device)->remotesockaddr, 
  273                      sizeof(((TgtSockStructT *) device)->remotesockaddr));
  274           /* if(n<0)
  275               printf("error writing to socket\n");
  276           */
  277       }
  278   }

  281   void TGT_libOutputs(void *device, char *err, void *src, int *nSamples)
  282   {
  283       int n = 0;
  284       if(device) {
  285           n = recvfrom(((TgtSockStructT *) device)->sockfd, src, 
  286                        ((TgtSockStructT *) device)->dataTypeSize * (*nSamples), 
  287                        0, NULL, NULL);
  288           if (n > 0) {
  289               *nSamples = n; /* Return number of bytes received */
  290           }
  291           else {
  292               *nSamples = 0; /* No UDP packets received */
  293           }
  294       }
  295   }
